import Foundation

let headers = ["end-user-token": "SOME_STRING_VALUE"]
let parameters = [
  "comment": [],
  "custom_id": [],
  "destination_city": [],
  "destination_state": [],
  "height": [],
  "length": [],
  "load_size": [],
  "origin_city": [],
  "origin_state": [],
  "other_trailer_types": [],
  "pay_rate": [],
  "posting_id": 0,
  "receive_date": [],
  "ship_date": [],
  "trailer_type": [],
  "weight": [],
  "width": []
] as [String : Any]

let postData = JSONSerialization.data(withJSONObject: parameters, options: [])

let request = NSMutableURLRequest(url: NSURL(string: "https://www.directfreight.com/api/postings/%7Bposting_type%7D")! as URL,
                                        cachePolicy: .useProtocolCachePolicy,
                                    timeoutInterval: 10.0)
request.httpMethod = "PATCH"
request.allHTTPHeaderFields = headers
request.httpBody = postData as Data

let session = URLSession.shared
let dataTask = session.dataTask(with: request as URLRequest, completionHandler: { (data, response, error) -> Void in
  if (error != nil) {
    print(error)
  } else {
    let httpResponse = response as? HTTPURLResponse
    print(httpResponse)
  }
})

dataTask.resume()
