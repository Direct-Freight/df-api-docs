  "content-type": "application/json",
  "accept": "application/json"
});

req.type("json");
req.send({
  "email": "<ADD STRING VALUE>",
  "end-user-token": "<ADD STRING VALUE>",
  "password": "<ADD STRING VALUE>"
});

});
